pipeline {
    agent {
        node {
            label 'prod'
        }
    }
    tools {
        maven 'mymaven'
    }

    stages {
        stage('code') {
            steps {
                git branch: 'master', url: 'https://github.com/venkatadeepthi79-svg/dockerwebapp.git'      
            }
        }
        stage('SonarQube Analysis') {
            steps {
            withSonarQubeEnv('sonarqube') {
                sh "mvn clean verify sonar:sonar -Dsonar.projectKey=devopsproject"
         }
    }   
}
        stage('Build using maven') {
            steps {
                sh 'mvn clean package'
                sh 'cp -r target Docker-app'
            }
        }
        stage('Build image') {
            steps {
                sh 'docker build -t venkatadeepthi/devopsproject:app1 Docker-app'
                sh 'docker build -t venkatadeepthi/devopsprojectdb:app2 Docker-db'
            }
        }
        stage('image scan') {
            steps {
                sh 'trivy image venkatadeepthi/devopsproject:app1'
                sh 'trivy image venkatadeepthi/devopsprojectdb:app2'
        }
    }
        stage('push image to Registry') {
            steps {
               withDockerRegistry(credentialsId: 'dockerhubcreds') {
                   sh 'docker push venkatadeepthi/devopsproject:app1'
                   sh 'docker push venkatadeepthi/devopsprojectdb:app2'
            }
        }
    }
        stage('Deploy') {
            steps {
                sh 'docker stack deploy -c compose.yml mystack'
            }
        }
       
    }

}
